/******************************************************************************
정수 A,B를 입력받아 다음 규칙에 따르는 프로그램을 작성하시오
1. A가 B의 배수이면 , A/B를 출력
2. A가 B보다 크고 배수가 아니라면 A % B 를 출력
3. A가 B보다 작다면, A + B를 출력
4. 아무것도 해당되지 않는다면 A,B 각각 출력
*******************************************************************************/

#include <stdio.h>

void main()
{
    int A, B;
    printf("정수 A를 입력하세요\n");
    scanf("%d", &A);
    printf("0이아닌 정수 B를 입력하세요\n");
    scanf("%d", &B);
    
    if (B == 0)
    {
        printf("0이아닌 정수 B를 입력해주세요");
        exit(1);
    }
    if (A % B ==0)
    {
        printf("1조건에 해당 A : %d B : %d", A / B);    
    } else if (A > B && A % B != 0) {
        printf("2조건에 해당 A %% B : %d", A % B); 
    } else if (A < B) {
        printf("3조건에 해당 A + B : %d", A + B); 
    } else {
        printf("아무것도 해당되지 않는다 A : %d B : %d", A, B);
    }
    
    printf("%d", 5 % 0);
}
