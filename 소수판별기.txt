def check_prime_num(x):
    for i in range(2, x):
        if x % i == 0:
            return False
    return True

number = input('판별할 자연수를 입력하세요:')
if not number.isdigit():
    print('자연수가 아닙니다.')
else:
    number = int(number)
    if number < 1:
        print('자연수가 아닙니다.')
    else:
        print('소수인가? ', check_prime_num(number))
